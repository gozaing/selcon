/* まず、postsテーブルを作成します: */
CREATE TABLE subjects (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(50),
    body TEXT,
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);

CREATE TABLE comments (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    commenter VARCHAR(50),
    body TEXT,
    subject_id INT,
    created DATETIME DEFAULT NULL,
    modified DATETIME DEFAULT NULL
);
